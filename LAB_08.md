FROM PRODUCT P
JOIN ORDER_POSITION OP ON OP.PRODUCT_ID=P.PRODUCT_ID
JOIN `ORDER` O ON O.ORDER_ID=OP.ORDER_ID
JOIN ORDER_STATUS OS ON OS.STATUS_ID=O.ORDER_STATUS
WHERE ORDER_STATUS in (3,10)

--- pkt4
```sql
SELECT C.CLIENT_ID, 
max(ORDER_DATE) as najnowsze, 
min(ORDER_DATE) as najwczesniejsze,
DATEDIFF(max(ORDER_DATE), min(ORDER_DATE))
FROM CLIENT C 
JOIN `ORDER` O ON O.CLIENT_ID=C.CLIENT_ID
GROUP by C.CLIENT_ID
```

--- pkt1
```sql
SELECT ORDER_DATE, 
CONCAT(YEAR(ORDER_DATE),' - ',WEEK(ORDER_DATE,1)) as ROK_TYDZIEN, 
CONCAT(YEAR(ORDER_DATE),' - ',DAYofyear(ORDER_DATE)) as ROK_DZIEN,
CONCAT(MONTHNAME(ORDER_DATE),'/',DAY(ORDER_DATE),'/',YEAR(ORDER_DATE)) as AGNIELSKA_NAZWA
FROM `ORDER` O
JOIN ORDER_STATUS S ON S.STATUS_ID=O.ORDER_STATUS
WHERE STATUS_ID in (3)
```
--- pkt2
```sql
SELECT O.ORDER_ID, SUM(AMOUNT*SELL_PRICE)
FROM ORDER_POSITION p
join `ORDER` O on p.ORDER_ID=O.ORDER_ID
where year(ORDER_DATE) = 2025
GROUP BY O.ORDER_ID
ORDER BY SUM(AMOUNT*SELL_PRICE) desc LIMIT 1
```
